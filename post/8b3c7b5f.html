切片与数组的区别&各⾃优势，forr 与 fori 的区别

数组

  • 固定⻓度：数组的⻓度在声明时就确定了，不能动态改变。
  • 值类型：数组是值类型，赋值或传参时会进⾏值拷⻉。
  • 性能：由于数组的⻓度固定，性能上会⽐切⽚更好。

切片

  • 动态⻓度：切⽚的⻓度可以动态改变，更加灵活。
  • 引⽤类型：切⽚是引⽤类型，赋值或传参时不会进⾏值拷⻉，⽽是引⽤同⼀个底层数组。
  • 内置函数⽀持：切⽚有很多内置函数⽀持，如 append、copy 等。

forr
range 提供了⼀种简洁的⽅式来迭代数组、切⽚、映射和通道。使⽤ range 迭代数组时，它会返回两个值：索引和该索引处的值。这使得你可以直接访问数组的元素，⽽不需要显式地使⽤索引来访问。## 指针

字符串函数

Contains（poc常用）

检查字符串是否包含⼦字符串。

  1. ContainsAny

检查字符串是否包含任何⼀个给定字符。

  1. ContainsRune
    检查字符串是否包含指定的 rune 字符。
  1. Count
    计算⼦字符串在字符串中出现的次数。
  1. EqualFold
    忽略⼤⼩写⽐较两个字符串是否相等。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.EqualFold(“Go”, “go”)) // true
    }
  2. Fields
    将字符串按空⽩字符分割成切⽚。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Fields(“hello world”)) // [“hello” “world”]
    }
  3. HasPrefix
    检查字符串是否以指定前缀开头。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.HasPrefix(“hello world”, “hello”)) // true
    }
  4. HasSuffix
    检查字符串是否以指定后缀结尾。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.HasSuffix(“hello world”, “world”)) // true
    }
  5. Index
    返回⼦字符串在字符串中第⼀次出现的位置。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Index(“hello world”, “world”)) // 6
    }
  6. IndexAny
    返回字符串中任意⼀个字符第⼀次出现的位置。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.IndexAny(“hello world”, “abc”)) // -1
    }
  7. IndexByte
    返回指定字节在字符串中第⼀次出现的位置。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.IndexByte(“hello world”, ‘o’)) // 4
    }
  8. IndexRune
    返回指定 rune 字符在字符串中第⼀次出现的位置。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.IndexRune(“hello world”, ‘w’)) // 6
    }
  9. Join
    将字符串切⽚⽤指定分隔符连接成⼀个字符串。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Join([]string{“hello”, “world”}, “, “)) //
    “hello, world”
    }
  10. LastIndex
    返回⼦字符串在字符串中最后⼀次出现的位置。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.LastIndex(“hello world”, “o”)) // 7
    }
  11. LastIndexAny
    返回字符串中任意⼀个字符最后⼀次出现的位置。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.LastIndexAny(“hello world”, “abc”)) // -1
    }
  12. Repeat
    重复字符串指定次数。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Repeat(“go”, 3)) // “gogogo”
    }
  13. Replace
    替换字符串中的⼦字符串。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Replace(“hello world”, “world”, “Go”, 1)) //
    “hello Go”
    }
  14. ReplaceAll
    替换字符串中的所有⼦字符串。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.ReplaceAll(“hello world world”, “world”, “Go”))
    // “hello Go Go”
    }
  15. Split
    将字符串按指定分隔符分割成切⽚。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Split(“hello,world”, “,”)) // [“hello” “world”]
    }
  16. SplitAfter
    将字符串按指定分隔符分割成切⽚，保留分隔符。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.SplitAfter(“hello,world”, “,”)) // [“hello,”
    “world”]
    }
  17. SplitAfterN
    将字符串按指定分隔符分割成切⽚，保留分隔符，最多分割 N 次。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.SplitAfterN(“a,b,c,d”, “,”, 2)) // [“a,”
    “b,c,d”]
    }
  18. SplitN
    将字符串按指定分隔符分割成切⽚，最多分割 N 次。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.SplitN(“a,b,c,d”, “,”, 2)) // [“a” “b,c,d”]
    }
  19. Title
    将字符串的每个单词的⾸字⺟⼤写。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Title(“hello world”)) // “Hello World”
    }
  20. ToLower
    将字符串转换为⼩写。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.ToLower(“HELLO WORLD”)) // “hello world”
    }
  21. ToUpper
    将字符串转换为⼤写。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.ToUpper(“hello world”)) // “HELLO WORLD”
    }
  22. Trim
    去除字符串两端的指定字符。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.Trim(“!!!hello!!!”, “!”)) // “hello”
    }
  23. TrimSpace
    去除字符串两端的空⽩字符。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.TrimSpace(“ hello “)) // “hello”
    }
  24. TrimPrefix
    去除字符串前缀。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.TrimPrefix(“hello world”, “hello “)) // “world”
    }
  25. TrimSuffix
    去除字符串后缀。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    fmt.Println(strings.TrimSuffix(“hello world”, “ world”)) // “hello”
    }
  26. NewReplacer
    创建⼀个字符串替换器。
    package main
    import (
    “fmt”
    “strings”
    )
    func main() {
    replacer := strings.NewReplacer(“world”, “Go”, “hello”, “Hi”)
    fmt.Println(replacer.Replace(“hello world”)) // “Hi Go”
    }

注意事项

⼤⼩写敏感：⼤多数函数都是⼤⼩写敏感的，如 Contains 、 Index 等。

空字符串处理：某些函数在处理空字符串时会返回特定结果，如 Split 会返回包含⼀个空字符串的切⽚。

Count 计算子字符串在字符串中出现的次数
EqualFold 忽略大小写比较两个字符串是否相等
Fields 将字符串按空白字符分割成切片
HasPrefix 检查字符串是否以指定前缀开头
HasSuffix 检查字符串是否以指定后缀结尾
Index 返回子字符串在字符串中第一次出现的位置
IndexAny 返回字符串中任何一个字符第一次出现的位置
IndexByte 返回指定字节在字符串中第一次出现的位置
IndexRune 返回指定 rune 字符在字符串中第一次出现的位置
Join 将字符串切片用指定分隔符连接成一个字符串
LastIndex 返回子字符串在字符串中最后一次出现的位置
LastIndexAny 返回字符串中任何一个字符最后一次出现的位置
Repeat 重复字符串指定次数
Replace 替换字符串中的子字符串
ReplaceAll 替换字符串中的所有子字符串

SplitAfterN 将字符串按指定分隔符分割成切片，保留分隔符，最多分割 N 次
SplitN 将字符串按指定分隔符分割成切片，最多分割 N 次
Title 将字符串的每个单词的首字母大写
ToLower 将字符串转换为小写
ToUpper 将字符串转换为大写
Trim 去除字符串两端的指定字符

TrimSuffix 去除字符串后缀
